#include <LiquidCrystal.h>
LiquidCrystal lcd (2,3,4,5,6,7);

int analogPin = 0;
int val = 0;
void setup ()
{
  Serial.begin(9600);
  lcd.begin(16,2);
  int analogPin = 0;
  int val = 0;
}
void loop()
{val=analogRead(analogPin);
 Serial.println(val);
 lcd.print("Hola");
 delay(3000);
 lcd.clear();
 val=analogRead(analogPin);
 Serial.println(val);
}
